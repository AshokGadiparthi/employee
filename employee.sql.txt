import java.text.ParseException;
import java.time.Instant;
import java.time.LocalDate;
import java.time.ZoneId;
import java.time.format.DateTimeFormatter;
import java.util.Date;

public class DateTimeConversion {
    public static void main(String[] args) {
        // Input date-time string in the "dd-MM-yyyy HH:mm:ss:SSSSSSSSS GMT" format
        String inputDateTimeString = "31-12-2022 15:30:45:123456789 GMT";

        // Define the formatter for the input format
        DateTimeFormatter inputFormatter = DateTimeFormatter.ofPattern("dd-MM-yyyy HH:mm:ss:SSSSSSSSS z");

        try {
            // Parse the input string to Instant
            Instant instant = Instant.from(inputFormatter.parse(inputDateTimeString));

            // Convert Instant to LocalDate
            LocalDate localDate = LocalDate.from(instant.atZone(ZoneId.of("GMT")));

            // Define the formatter for the output format
            DateTimeFormatter outputFormatter = DateTimeFormatter.ofPattern("yyyy-MM-dd");

            // Format LocalDate to the desired output format
            String outputDateString = localDate.format(outputFormatter);

            // Print the result
            System.out.println("Input Date-Time: " + inputDateTimeString);
            System.out.println("Output Date: " + outputDateString);
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
